74       private void gerarRelatorioPDF(String mes) {
75           try {
76               Document document = new Document();
77               String nomeArquivo = "relatorio_" + mes + ".pdf";
78               File file = new File(getExternalFilesDir(null), nomeArquivo);
79               PdfWriter.getInstance(document, new FileOutputStream(file));
80               document.open();
81   
82               // Adicione o conteúdo do relatório ao documento
83               document.add(new Paragraph("Relatório Mensal de Tráfego - " + mes));
84   
85               // Obtenha os dados do banco de dados
86               SQLiteDatabase db = bancoDeDados.getReadableDatabase();
87               Cursor cursor = db.query("bdts", null, "strftime('%m/%Y', data) = ?", new String[]{mes}, null, null, null);
88   
89               if (cursor.moveToFirst()) {
90                   do {
91                       // Adicione os dados de cada BDT ao relatório
92                       document.add(new Paragraph("Data: " + cursor.getString(cursor.getColumnIndex("data"))));
93                       document.add(new Paragraph("Percurso/Destino: " + cursor.getString(cursor.getColumnIndex("percurso_destino"))));
94                       // Adicione outros dados do BDT
95                       document.add(new Paragraph("-----------------------------------"));
96                   } while (cursor.moveToNext());
97               }
98   
99               cursor.close();
100              db.close();
101  
102              document.close();
103  
104              Toast.makeText(this, "Relatório gerado com sucesso: " + file.getAbsolutePath(), Toast.LENGTH_LONG).show();
105          } catch (Exception e) {
106              e.printStackTrace();
107              Toast.makeText(this, "Erro ao gerar relatório", Toast.LENGTH_SHORT).show();
108          }
109      }

111      private void enviarEmail(String destinatario, String assunto, String corpo, String caminhoArquivo) {
112          new Thread(new Runnable() {
113              @Override
114              public void run() {
115                  try {
116                      // Configurações do e-mail
117                      Properties props = new Properties();
118                      props.put("mail.smtp.host", "smtp.gmail.com"); // Use o servidor SMTP do seu provedor de e-mail
119                      props.put("mail.smtp.socketFactory.port", "465");
120                      props.put("mail.smtp.socketFactory.class", "javax.net.ssl.SSLSocketFactory");
121                      props.put("mail.smtp.auth", "true");
122                      props.put("mail.smtp.port", "465");
123  
124                      Session session = Session.getDefaultInstance(props,
125                              new javax.mail.Authenticator() {
126                                  protected PasswordAuthentication getPasswordAuthentication() {
127                                      return new PasswordAuthentication("seu_email@gmail.com", "sua_senha"); // Substitua pelo seu e-mail e senha
128                                  }
129                              });
130  
131                      Message message = new MimeMessage(session);
132                      message.setFrom(new InternetAddress("seu_email@gmail.com")); // Substitua pelo seu e-mail
133                      message.setRecipients(Message.RecipientType.TO, InternetAddress.parse(destinatario));
134                      message.setSubject(assunto);
135  
136                      // Cria a mensagem com o corpo e o anexo
137                      BodyPart messageBodyPart = new MimeBodyPart();
138                      messageBodyPart.setText(corpo);
139  
140                      Multipart multipart = new MimeMultipart();
141                      multipart.addBodyPart(messageBodyPart);
142  
143                      messageBodyPart = new MimeBodyPart();
144                      DataSource source = new FileDataSource(caminhoArquivo);
145                      messageBodyPart.setDataHandler(new DataHandler(source));
146                      messageBodyPart.setFileName(new File(caminhoArquivo).getName());
147                      multipart.addBodyPart(messageBodyPart);
148  
149                      message.setContent(multipart);
150  
151                      // Envia o e-mail
152                      Transport.send(message);
153  
154                      runOnUiThread(new Runnable() {
155                          @Override
156                          public void run() {
157                              Toast.makeText(RelatoriosActivity.this, "E-mail enviado com sucesso", Toast.LENGTH_SHORT).show();
158                          }
159                      });
160                  } catch (MessagingException e) {
161                      e.printStackTrace();
162                      runOnUiThread(new Runnable() {
163                          @Override
164                          public void run() {
165                              Toast.makeText(RelatoriosActivity.this, "Erro ao enviar e-mail", Toast.LENGTH_SHORT).show();
166                          }
167                      });
168                  }
169              }
170          }).start();
171      }
